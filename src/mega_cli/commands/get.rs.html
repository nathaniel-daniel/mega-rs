26fn path_ends_with_sep(path: &Path) -> bool {
27    path.as_os_str()
28        .as_encoded_bytes()
29        .last()
30        .is_some_and(|b| std::path::is_separator(char::from(*b)))
31}

41async fn download_file(
42    client: &mega::EasyClient,
43    args: DownloadFileArgs,
44    output: Option<&Path>,
45) -> anyhow::Result<()> {
46    let mut builder = mega::EasyGetAttributesBuilder::new();
47    builder.include_download_url(true);
48    if let Some(public_node_id) = args.public_node_id {
49        builder.public_node_id(public_node_id);
50    }
51    if let Some(node_id) = args.node_id {
52        builder.node_id(node_id);
53    }
54    if let Some(reference_node_id) = args.reference_node_id {
55        builder.reference_node_id(reference_node_id);
56    }
57
58    let attributes = client
59        .get_attributes(builder)
60        .await
61        .context("failed to get attributes")?;
62    let decoded_attributes = attributes
63        .decode_attributes(args.file_key.key)
64        .context("failed to decode attributes")?;
65    let download_url = attributes
66        .download_url
67        .as_ref()
68        .context("missing download url")?;
69
70    let output = match output.as_ref() {
71        Some(output) => {
72            if path_ends_with_sep(output) {
73                output.join(decoded_attributes.name)
74            } else {
75                output.to_path_buf()
76            }
77        }
78        None => PathBuf::from(decoded_attributes.name),
79    };
80
81    if tokio::fs::try_exists(&output)
82        .await
83        .with_context(|| format!("failed to check if \"{}\" exists", output.display()))?
84    {
85        let file_key = args.file_key.clone();
86        let output = output.clone();
87        tokio::task::spawn_blocking(move || {
88            use std::fs::File;
89            use std::io::Read;
90
91            let mut validator = mega::FileValidator::new(file_key);
92            let mut buffer = vec![0; 16 * 1024];
93            let mut file = File::open(&output)?;
94            loop {
95                let n = file.read(&mut buffer)?;
96                if n == 0 {
97                    break;
98                }
99                validator.feed(&buffer[..n]);
100            }
101
102            // TODO: Catch error here, display pretty error, or maybe delete?
103            validator.finish()?;
104
105            anyhow::Ok(())
106        })
107        .await??;
108
109        // If we reach here without erroring, the file was already downloaded.
110        return Ok(());
111    }
112
113    let temp_output = output.with_added_extension("temp");
114    let mut output_file = File::create(&temp_output)
115        .await
116        .with_context(|| format!("failed to open \"{}\"", temp_output.display()))?;
117    let mut reader = client
118        .download_file(&args.file_key, download_url.as_str())
119        .await
120        .context("failed to get download stream")?;
121
122    let progress_bar = indicatif::ProgressBar::new(attributes.size);
123    let progress_bar_style_template = "[Time = {elapsed_precise} | ETA = {eta_precise} | Speed = {bytes_per_sec}] {wide_bar} {bytes}/{total_bytes}";
124    let progress_bar_style = indicatif::ProgressStyle::default_bar()
125        .template(progress_bar_style_template)
126        .expect("invalid progress bar style template");
127    progress_bar.set_style(progress_bar_style);
128
129    let progress_bar_tick_handle = {
130        let progress_bar = progress_bar.clone();
131        tokio::spawn(async move {
132            while !progress_bar.is_finished() {
133                progress_bar.tick();
134                tokio::time::sleep(Duration::from_millis(1_000)).await;
135            }
136        })
137    };
138    tokio::io::copy(
139        &mut progress_bar.wrap_async_read(&mut reader),
140        &mut output_file,
141    )
142    .await?;
143    output_file.flush().await?;
144    output_file.sync_all().await?;
145    tokio::fs::rename(temp_output, output).await?;
146    progress_bar.finish();
147
148    progress_bar_tick_handle.await?;
149
150    Ok(())
151}

168async fn download_folder(
169    client: &mega::EasyClient,
170    mut args: DownloadFolderArgs,
171    output: Option<&Path>,
172) -> anyhow::Result<()> {
173    let output = match output.as_ref() {
174        Some(output) => {
175            if path_ends_with_sep(output) {
176                output.join(&args.root_folder_name)
177            } else {
178                output.to_path_buf()
179            }
180        }
181        None => PathBuf::from(&args.root_folder_name),
182    };
183
184    let temp_output = output.with_added_extension("temp");
185    tokio::fs::create_dir_all(&temp_output)
186        .await
187        .context("failed to create temp folder")?;
188
189    let mut id_to_path = HashMap::new();
190    id_to_path.insert(args.root_folder_id.clone(), temp_output.clone());
191    let mut stack = vec![args.root_folder_id.clone()];
192    while let Some(parent_node_id) = stack.pop() {
193        let children = match args.children.remove(&parent_node_id) {
194            Some(children) => children,
195            None => {
196                continue;
197            }
198        };
199        let parent_path = id_to_path
200            .get(&parent_node_id)
201            .context("missing parent path")?
202            .clone();
203
204        for child in children.into_iter() {
205            // Windows does not allow file names to end with a trailing slash,
206            // even though other platforms do.
207            // (Windows actuallly allows this through UNC paths, but this breaks the file explorer,
208            // and official documentation says not to create files with trailing spaces.)
209            //
210            // If we only sanitize on Windows, downloaded folders will be different between platforms.
211            // This can be a problem if a user downloads a file on Windows, and downloads again on another platform in the same folder.
212            // It is a desirable feature to have the same folder structure on all platforms.
213            //
214            // As a result, perform this sanitization on all platforms.
215            // TODO: Ensure sanitizing here doesn't clobber paths.
216            let child_name = {
217                let mut value = child.name.clone();
218                while value.ends_with(' ') {
219                    value.pop();
220                }
221                value
222            };
223
224            let node_path = parent_path.join(child_name);
225            let file_key = match child.key {
226                mega::FileOrFolderKey::File(key) => key,
227                mega::FileOrFolderKey::Folder(_key) => {
228                    tokio::fs::create_dir_all(&node_path).await?;
229
230                    id_to_path.insert(child.id.clone(), node_path);
231                    stack.push(child.id);
232                    continue;
233                }
234            };
235
236            download_file(
237                client,
238                DownloadFileArgs {
239                    file_key,
240                    node_id: Some(child.id),
241                    public_node_id: None,
242                    reference_node_id: Some(args.reference_node_id.clone()),
243                },
244                Some(&node_path),
245            )
246            .await?;
247        }
248    }
249
250    tokio::fs::rename(&temp_output, &output)
251        .await
252        .context("failed to rename temp folder")?;
253
254    Ok(())
255}

257#[derive(Debug)]
258enum DownloadType {
259    File(DownloadFileArgs),
260    Folder(DownloadFolderArgs),
261}

263async fn parse_input_arg(
264    client: &mega::EasyClient,
265    input: &str,
266    options_file_key: Option<mega::FileKey>,
267) -> anyhow::Result<DownloadType> {
268    let mut public_node_id = None;
269    let mut node_id = None;
270    let mut file_key = None;
271    let mut reference_node_id = None;
272
273    // If it starts with a url, assume it's a url.
274    // Otherwise, assume it's a raw id.
275    if input.starts_with("https://mega.nz") {
276        let url = Url::parse(input).context("invalid url")?;
277        let parsed_url = mega::ParsedMegaUrl::try_from(&url).context("failed to parse mega url")?;
278
279        match parsed_url {
280            mega::ParsedMegaUrl::File(file_url) => {
281                public_node_id = Some(file_url.file_id.to_string());
282                file_key = Some(file_url.file_key.clone());
283            }
284            mega::ParsedMegaUrl::Folder(folder_url) => {
285                reference_node_id = Some(folder_url.folder_id.clone());
286
287                let fetch_nodes_response = client
288                    .fetch_nodes(Some(&folder_url.folder_id), true)
289                    .await?;
290
291                let child_data = match folder_url.child_data.as_ref() {
292                    Some(child_data) => child_data,
293                    None => {
294                        let mut children = HashMap::with_capacity(fetch_nodes_response.nodes.len());
295                        let mut root_folder_name = None;
296                        let mut root_folder_id = None;
297                        for node_entry in fetch_nodes_response.nodes.iter() {
298                            let node_decoded_attributes =
299                                node_entry.decode_attributes(&folder_url.folder_key)?;
300
301                            let (root_id, _) = node_entry
302                                .key
303                                .split_once(':')
304                                .context("invalid key format")?;
305                            if root_id == node_entry.id {
306                                ensure!(root_folder_name.is_none());
307                                root_folder_name = Some(node_decoded_attributes.name);
308                                root_folder_id = Some(root_id.to_string());
309                                continue;
310                            }
311                            let key = node_entry.decrypt_key(&folder_url.folder_key)?;
312
313                            children
314                                .entry(node_entry.parent_id.clone())
315                                .or_insert(Vec::new())
316                                .push(DownloadFolderArgsChild {
317                                    name: node_decoded_attributes.name,
318                                    id: node_entry.id.clone(),
319                                    key,
320                                });
321                        }
322
323                        return Ok(DownloadType::Folder(DownloadFolderArgs {
324                            root_folder_name: root_folder_name
325                                .context("missing root folder node entry")?,
326                            root_folder_id: root_folder_id
327                                .context("missing root folder node entry")?,
328                            reference_node_id: folder_url.folder_id.clone(),
329                            children,
330                        }));
331                    }
332                };
333                ensure!(
334                    child_data.is_file,
335                    "child folder downloads are currently unsupported"
336                );
337
338                let node_entry = fetch_nodes_response
339                    .nodes
340                    .iter()
341                    .find(|node| node.id == child_data.node_id)
342                    .context("missing file node in folder listing")?;
343                let node_key = node_entry
344                    .decrypt_key(&folder_url.folder_key)?
345                    .take_file_key()
346                    .context("folder downloads are currently unsupported")?;
347
348                node_id = Some(child_data.node_id.to_string());
349                file_key = Some(node_key.clone());
350            }
351        }
352    } else {
353        node_id = Some(input.to_string());
354    };
355
356    let file_key = options_file_key.or(file_key).context("missing file key")?;
357
358    Ok(DownloadType::File(DownloadFileArgs {
359        file_key,
360        public_node_id,
361        node_id,
362        reference_node_id,
363    }))
364}

366pub async fn exec(client: &mega::EasyClient, options: &Options) -> anyhow::Result<()> {
367    let options_file_key = options
368        .key
369        .as_deref()
370        .map(|file_key| file_key.parse::<mega::FileKey>())
371        .transpose()?;
372    let download_type = parse_input_arg(client, &options.input, options_file_key).await?;
373
374    match download_type {
375        DownloadType::File(mut args) => {
376            if let Some(reference_node_id) = options.reference_node_id.clone() {
377                args.reference_node_id = Some(reference_node_id);
378            }
379
380            download_file(client, args, options.output.as_deref()).await?;
381        }
382        DownloadType::Folder(args) => {
383            download_folder(client, args, options.output.as_deref()).await?;
384        }
385    }
386
387    Ok(())
